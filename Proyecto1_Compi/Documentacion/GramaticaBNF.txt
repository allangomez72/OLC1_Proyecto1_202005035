<INICIO> ::= <INSTRUCTIONS>

<INSTRUCTIONS> ::= PROGRAM <INSTRUCTION> END PROGRAM
                 | error END PROGRAM

<INSTRUCTION> ::= <INSTRUCTION> <SENTENCE>
                | <SENTENCE>

<SENTENCE> ::= <STATEMENT_VARIABLES>
             | <STATEMENT_PRINT>
             | <ARREGLO_DECLARATION>
             | <PRINT_ARRAY>
             | <FUNCTION_GRAPH>

<STATEMENT_VARIABLES> ::= VAR DOSPUNTOS <TYPE_VAR> DOSPUNTOS DOSPUNTOS ID FLECHA <EXPRESSION> END PUNTOCOMA

<TYPE_VAR> ::= CHAR
             | DOUBLE

<EXPRESSION> ::= DECIMAL
               | ID
               | CADENA
               | SUMA PARABRE <EXPRESSION> COMA <EXPRESSION> PARCIERRA
               | RESTA PARABRE <EXPRESSION> COMA <EXPRESSION> PARCIERRA
               | MULTIPLICACION PARABRE <EXPRESSION> COMA <EXPRESSION> PARCIERRA
               | DIVISION PARABRE <EXPRESSION> COMA <EXPRESSION> PARCIERRA
               | MODULO PARABRE <EXPRESSION> COMA <EXPRESSION> PARCIERRA
               | MEDIA PARABRE <TYPE_PRINT_ARRAY> PARCIERRA
               | MEDIANA PARABRE <TYPE_PRINT_ARRAY> PARCIERRA
               | MODA PARABRE <TYPE_PRINT_ARRAY> PARCIERRA
               | VARIANZA PARABRE <TYPE_PRINT_ARRAY> PARCIERRA
               | MAX PARABRE <TYPE_PRINT_ARRAY> PARCIERRA
               | MIN PARABRE <TYPE_PRINT_ARRAY> PARCIERRA

<STATEMENT_PRINT> ::= CONSOLE DOSPUNTOS DOSPUNTOS PRINT IGUAL <CONCATENATION> END PUNTOCOMA

<CONCATENATION> ::= <EXPRESSION> COMA <CONCATENATION>
                  | <EXPRESSION>

<ARREGLO_DECLARATION> ::= ARREGLO DOSPUNTOS <ARREGLO_TYPE> DOSPUNTOS DOSPUNTOS ARROBA ID FLECHA CORABRE <LIST_VALUES> CORCIERRA END PUNTOCOMA

<ARREGLO_TYPE> ::= TYPE CORABRE CORCIERRA
                 | TYPE

<LIST_VALUES> ::= <EXPRESSION> COMA <LIST_VALUES>
                | <EXPRESSION>

<PRINT_ARRAY> ::= CONSOLE DOSPUNTOS DOSPUNTOS COLUMN IGUAL <EXPRESSION> FLECHA <TYPE_PRINT_ARRAY> END PUNTOCOMA

<TYPE_PRINT_ARRAY> ::= ARROBA ID
                      | CORABRE <LIST_VALUES> CORCIERRA

<FUNCTION_GRAPH> ::= GRAPH <TYPE_GRAPH> PARABRE <INSTRUCTION_GRAPH> PARCIERRA END PUNTOCOMA

<TYPE_GRAPH> ::= PRBAR 
               | PRPIE 
               | PRLINE 
               | HISTOGRAM

<EXPRESSSION_GRAPH> ::= <TITULO_PRINCIPAL>
                       | <TITULO_EJE>
                       | <EJEES_VALORES>
                       | <LABELS>
                       | <VALORES>
                       | <EXEC_GRAPH>

<TITULO_PRINCIPAL> ::= TITULO DOSPUNTOS DOSPUNTOS <TYPE_VAR> IGUAL <EXPRESSION> END PUNTOCOMA

<TITULO_EJE> ::= TITULO <TYPE_EJE> DOSPUNTOS DOSPUNTOS <TYPE_VAR> IGUAL <EXPRESSION> END PUNTOCOMA

<EJEES_VALORES> ::= EJE <TYPE_EJE> DOSPUNTOS DOSPUNTOS <TYPE_VAR> IGUAL <TYPE_PRINT_ARRAY> END PUNTOCOMA

<LABELS> ::= LABEL DOSPUNTOS DOSPUNTOS <TYPE_VAR> IGUAL <TYPE_PRINT_ARRAY> END PUNTOCOMA

<VALORES> ::= VALUES DOSPUNTOS DOSPUNTOS <TYPE_VAR> IGUAL <TYPE_PRINT_ARRAY> END PUNTOCOMA

<EXEC_GRAPH> ::= EXEC GRAPH <TYPE_GRAPH> END PUNTOCOMA

<TYPE_EJE> ::= PRX 
             | PRY
